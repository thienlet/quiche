47#[derive(Debug)]
48struct SentPacket {
49    pkt_num: u64,
50    status: SentStatus,
51}
52
53#[derive(Debug)]
54enum SentStatus {
55    Sent {
56        time_sent: Instant,
57        ack_eliciting: bool,
58        in_flight: bool,
59        has_data: bool,
60        pmtud: bool,
61        sent_bytes: usize,
62        frames: SmallVec<[frame::Frame; 1]>,
63    },
64    Acked,
65    Lost,
66}
67
68impl SentStatus {
69    fn ack(&mut self) -> Self {
70        std::mem::replace(self, SentStatus::Acked)
71    }
72
73    fn lose(&mut self) -> Self {
74        if !matches!(self, SentStatus::Acked) {
75            std::mem::replace(self, SentStatus::Lost)
76        } else {
77            SentStatus::Acked
78        }
79    }
80}

121impl RecoveryEpoch {
122    /// Discard the Epoch state and return the total size of unacked packets
123    /// that were discarded
124    fn discard(&mut self, cc: &mut impl CongestionControl) -> usize {
125        let unacked_bytes = self
126            .sent_packets
127            .drain(..)
128            .map(|p| {
129                if let SentPacket {
130                    status:
131                        SentStatus::Sent {
132                            in_flight,
133                            sent_bytes,
134                            ..
135                        },
136                    pkt_num,
137                } = p
138                {
139                    cc.on_packet_neutered(pkt_num);
140                    if in_flight {
141                        return sent_bytes;
142                    }
143                }
144                0
145            })
146            .sum();
147
148        std::mem::take(&mut self.sent_packets);
149        self.time_of_last_ack_eliciting_packet = None;
150        self.loss_time = None;
151        self.loss_probes = 0;
152        self.pkts_in_flight = 0;
153
154        unacked_bytes
155    }
156
157    fn detect_and_remove_acked_packets(
158        &mut self, acked: &RangeSet, newly_acked: &mut Vec<Acked>, trace_id: &str,
159    ) -> AckedDetectionResult {
160        // Update the largest acked packet
161        self.largest_acked_packet.replace(
162            self.largest_acked_packet
163                .unwrap_or(0)
164                .max(acked.last().unwrap()),
165        );
166
167        newly_acked.clear();
168
169        let mut acked_bytes = 0;
170        let mut spurious_losses = 0;
171        let mut spurious_pkt_thresh = None;
172        let mut has_ack_eliciting = false;
173
174        let largest_acked = self.largest_acked_packet.unwrap();
175
176        for ack in acked.iter() {
177            // Because packets always have incrementing numbers, they are always
178            // in sorted order.
179            let start = if self
180                .sent_packets
181                .front()
182                .filter(|e| e.pkt_num >= ack.start)
183                .is_some()
184            {
185                // Usually it will be the first packet.
186                0
187            } else {
188                self.sent_packets
189                    .binary_search_by_key(&ack.start, |p| p.pkt_num)
190                    .unwrap_or_else(|e| e)
191            };
192
193            for SentPacket { pkt_num, status } in
194                self.sent_packets.range_mut(start..)
195            {
196                if *pkt_num < ack.end {
197                    match status.ack() {
198                        SentStatus::Sent {
199                            time_sent,
200                            in_flight,
201                            sent_bytes,
202                            frames,
203                            ack_eliciting,
204                            ..
205                        } => {
206                            if in_flight {
207                                self.pkts_in_flight -= 1;
208                                acked_bytes += sent_bytes;
209                            }
210                            newly_acked.push(Acked {
211                                pkt_num: *pkt_num,
212                                time_sent,
213                            });
214
215                            self.acked_frames.extend(frames);
216
217                            has_ack_eliciting |= ack_eliciting;
218
219                            trace!("{} packet newly acked {}", trace_id, pkt_num);
220                        },
221
222                        SentStatus::Acked => {},
223                        SentStatus::Lost => {
224                            // An acked packet was already declared lost
225                            spurious_losses += 1;
226                            spurious_pkt_thresh
227                                .get_or_insert(largest_acked - *pkt_num + 1);
228                        },
229                    }
230                } else {
231                    break;
232                }
233            }
234        }
235
236        self.drain_acked_and_lost_packets();
237
238        AckedDetectionResult {
239            acked_bytes,
240            spurious_losses,
241            spurious_pkt_thresh,
242            has_ack_eliciting,
243        }
244    }

246    fn detect_and_remove_lost_packets(
247        &mut self, loss_delay: Duration, pkt_thresh: u64, now: Instant,
248        newly_lost: &mut Vec<Lost>,
249    ) -> LossDetectionResult {
250        newly_lost.clear();
251        let mut lost_bytes = 0;
252        self.loss_time = None;
253
254        let lost_send_time = now.checked_sub(loss_delay).unwrap();
255        let largest_acked = self.largest_acked_packet.unwrap_or(0);
256        let mut pmtud_lost_bytes = 0;
257        let mut pmtud_lost_packets = SmallVec::new();
258
259        for SentPacket { pkt_num, status } in &mut self.sent_packets {
260            if *pkt_num > largest_acked {
261                break;
262            }
263
264            if let SentStatus::Sent { time_sent, .. } = status {
265                if *time_sent <= lost_send_time ||
266                    largest_acked >= *pkt_num + pkt_thresh
267                {
268                    if let SentStatus::Sent {
269                        in_flight,
270                        sent_bytes,
271                        frames,
272                        pmtud,
273                        ..
274                    } = status.lose()
275                    {
276                        self.lost_frames.extend(frames);
277
278                        if in_flight {
279                            self.pkts_in_flight -= 1;
280
281                            if pmtud {
282                                pmtud_lost_bytes += sent_bytes;
283                                pmtud_lost_packets.push(*pkt_num);
284                                // Do not track PMTUD probes losses
285                                continue;
286                            }
287
288                            lost_bytes += sent_bytes;
289                        }
290
291                        newly_lost.push(Lost {
292                            packet_number: *pkt_num,
293                            bytes_lost: sent_bytes,
294                        });
295                    }
296                } else {
297                    self.loss_time = Some(*time_sent + loss_delay);
298                    break;
299                }
300            }
301        }
302
303        LossDetectionResult {
304            lost_bytes,
305            lost_packets: newly_lost.len(),
306
307            pmtud_lost_bytes,
308            pmtud_lost_packets,
309        }
310    }
311
312    /// Remove packets that were already handled from the front of the queue,
313    /// but avoid removing packets from the middle of the queue to avoid
314    /// compaction
315    fn drain_acked_and_lost_packets(&mut self) {
316        while let Some(SentPacket {
317            status: SentStatus::Acked | SentStatus::Lost,
318            ..
319        }) = self.sent_packets.front()
320        {
321            self.sent_packets.pop_front();
322        }
323    }
324
325    fn least_unacked(&self) -> u64 {
326        for pkt in self.sent_packets.iter() {
327            if let SentPacket {
328                pkt_num,
329                status: SentStatus::Sent { .. },
330            } = pkt
331            {
332                return *pkt_num;
333            }
334        }
335
336        self.largest_acked_packet.unwrap_or(0) + 1
337    }
338}

431    fn detect_and_remove_lost_packets(
432        &mut self, epoch: packet::Epoch, now: Instant,
433    ) -> (usize, usize) {
434        let lost = &mut self.lost_reuse;
435
436        let LossDetectionResult {
437            lost_bytes,
438            lost_packets,
439
440            pmtud_lost_bytes,
441            pmtud_lost_packets,
442        } = self.epochs[epoch].detect_and_remove_lost_packets(
443            self.rtt_stats.loss_delay(self.time_thresh),
444            self.pkt_thresh,
445            now,
446            lost,
447        );
448
449        self.bytes_in_flight -= lost_bytes + pmtud_lost_bytes;
450
451        for pkt in pmtud_lost_packets {
452            self.pacer.on_packet_neutered(pkt);
453        }
454
455        (lost_bytes, lost_packets)
456    }

458    fn loss_time_and_space(&self) -> (Option<Instant>, packet::Epoch) {
459        let mut epoch = packet::Epoch::Initial;
460        let mut time = self.epochs[epoch].loss_time;
461
462        // Iterate over all packet number spaces starting from Handshake.
463        for e in [packet::Epoch::Handshake, packet::Epoch::Application] {
464            let new_time = self.epochs[e].loss_time;
465            if time.is_none() || new_time < time {
466                time = new_time;
467                epoch = e;
468            }
469        }
470
471        (time, epoch)
472    }
473
474    fn pto_time_and_space(
475        &self, handshake_status: HandshakeStatus, now: Instant,
476    ) -> (Option<Instant>, packet::Epoch) {
477        let mut duration = self.pto() * (1 << self.pto_count);
478
479        // Arm PTO from now when there are no inflight packets.
480        if self.bytes_in_flight == 0 {
481            if handshake_status.has_handshake_keys {
482                return (Some(now + duration), packet::Epoch::Handshake);
483            } else {
484                return (Some(now + duration), packet::Epoch::Initial);
485            }
486        }
487
488        let mut pto_timeout = None;
489        let mut pto_space = packet::Epoch::Initial;
490
491        // Iterate over all packet number spaces.
492        for &e in packet::Epoch::epochs(
493            packet::Epoch::Initial..=packet::Epoch::Application,
494        ) {
495            if self.epochs[e].pkts_in_flight == 0 {
496                continue;
497            }
498
499            if e == packet::Epoch::Application {
500                // Skip Application Data until handshake completes.
501                if !handshake_status.completed {
502                    return (pto_timeout, pto_space);
503                }
504
505                // Include max_ack_delay and backoff for Application Data.
506                duration +=
507                    self.rtt_stats.max_ack_delay * 2_u32.pow(self.pto_count);
508            }
509
510            let new_time = self.epochs[e]
511                .time_of_last_ack_eliciting_packet
512                .map(|t| t + duration);
513
514            if pto_timeout.is_none() || new_time < pto_timeout {
515                pto_timeout = new_time;
516                pto_space = e;
517            }
518        }
519
520        (pto_timeout, pto_space)
521    }
522
523    fn set_loss_detection_timer(
524        &mut self, handshake_status: HandshakeStatus, now: Instant,
525    ) {
526        if let (Some(earliest_loss_time), _) = self.loss_time_and_space() {
527            // Time threshold loss detection.
528            self.loss_timer.update(earliest_loss_time);
529            return;
530        }
531
532        if self.bytes_in_flight == 0 && handshake_status.peer_verified_address {
533            self.loss_timer.clear();
534            return;
535        }
536
537        // PTO timer.
538        if let (Some(timeout), _) = self.pto_time_and_space(handshake_status, now)
539        {
540            self.loss_timer.update(timeout);
541        }
542    }

590    fn on_packet_sent(
591        &mut self, pkt: Sent, epoch: packet::Epoch,
592        handshake_status: HandshakeStatus, now: Instant, trace_id: &str,
593    ) {
594        let time_sent = self.get_next_release_time().time(now).unwrap_or(now);
595
596        let epoch = &mut self.epochs[epoch];
597
598        let ack_eliciting = pkt.ack_eliciting;
599        let in_flight = pkt.in_flight;
600        let sent_bytes = pkt.size;
601        let pkt_num = pkt.pkt_num;
602
603        if let Some(SentPacket { pkt_num, .. }) = epoch.sent_packets.back() {
604            assert!(*pkt_num < pkt.pkt_num, "Packet numbers must increase");
605        }
606
607        let status = SentStatus::Sent {
608            time_sent,
609            ack_eliciting,
610            in_flight,
611            pmtud: pkt.pmtud,
612            has_data: pkt.has_data,
613            sent_bytes,
614            frames: pkt.frames,
615        };
616
617        epoch.sent_packets.push_back(SentPacket { pkt_num, status });
618
619        if ack_eliciting {
620            epoch.time_of_last_ack_eliciting_packet = Some(time_sent);
621            self.outstanding_non_ack_eliciting = 0;
622        } else {
623            self.outstanding_non_ack_eliciting += 1;
624        }
625
626        if in_flight {
627            self.pacer.on_packet_sent(
628                time_sent,
629                self.bytes_in_flight,
630                pkt_num,
631                sent_bytes,
632                pkt.has_data,
633                &self.rtt_stats,
634            );
635
636            self.bytes_in_flight += sent_bytes;
637            epoch.pkts_in_flight += 1;
638            self.set_loss_detection_timer(handshake_status, time_sent);
639        }
640
641        self.bytes_sent += sent_bytes;
642
643        trace!("{} {:?}", trace_id, self);
644    }

648    }

650    fn on_ack_received(
651        &mut self, ranges: &RangeSet, ack_delay: u64, epoch: packet::Epoch,
652        handshake_status: HandshakeStatus, now: Instant, trace_id: &str,
653    ) -> OnAckReceivedOutcome {
654        let prior_in_flight = self.bytes_in_flight;
655
656        let AckedDetectionResult {
657            acked_bytes,
658            spurious_losses,
659            spurious_pkt_thresh,
660            has_ack_eliciting,
661        } = self.epochs[epoch].detect_and_remove_acked_packets(
662            ranges,
663            &mut self.newly_acked,
664            trace_id,
665        );
666
667        self.lost_spurious_count += spurious_losses;
668        if let Some(thresh) = spurious_pkt_thresh {
669            self.pkt_thresh =
670                self.pkt_thresh.max(thresh.min(MAX_PACKET_THRESHOLD));
671        }
672
673        if self.newly_acked.is_empty() {
674            return OnAckReceivedOutcome::default();
675        }
676
677        self.bytes_in_flight -= acked_bytes;
678
679        // Check if largest packet is newly acked.
680        let largest_newly_acked = self.newly_acked.last().unwrap();
681        let update_rtt: bool = largest_newly_acked.pkt_num ==
682            ranges.last().unwrap() &&
683            has_ack_eliciting;
684        if update_rtt {
685            let latest_rtt = now - largest_newly_acked.time_sent;
686            self.rtt_stats.update_rtt(
687                latest_rtt,
688                Duration::from_micros(ack_delay),
689                now,
690                handshake_status.completed,
691            );
692        }
693
694        let (lost_bytes, lost_packets) =
695            self.detect_and_remove_lost_packets(epoch, now);
696
697        self.pacer.on_congestion_event(
698            update_rtt,
699            prior_in_flight,
700            self.bytes_in_flight,
701            now,
702            &self.newly_acked,
703            &self.lost_reuse,
704            self.epochs[epoch].least_unacked(),
705            &self.rtt_stats,
706            &mut self.recovery_stats,
707        );
708
709        self.pto_count = 0;
710        self.lost_count += lost_packets;
711
712        self.set_loss_detection_timer(handshake_status, now);
713
714        trace!("{} {:?}", trace_id, self);
715
716        OnAckReceivedOutcome {
717            lost_packets,
718            lost_bytes,
719            acked_bytes,
720            spurious_losses,
721        }
722    }

724    fn on_loss_detection_timeout(
725        &mut self, handshake_status: HandshakeStatus, now: Instant,
726        trace_id: &str,
727    ) -> OnLossDetectionTimeoutOutcome {
728        let (earliest_loss_time, epoch) = self.loss_time_and_space();
729
730        if earliest_loss_time.is_some() {
731            let prior_in_flight = self.bytes_in_flight;
732
733            let (lost_bytes, lost_packets) =
734                self.detect_and_remove_lost_packets(epoch, now);
735
736            self.pacer.on_congestion_event(
737                false,
738                prior_in_flight,
739                self.bytes_in_flight,
740                now,
741                &[],
742                &self.lost_reuse,
743                self.epochs[epoch].least_unacked(),
744                &self.rtt_stats,
745                &mut self.recovery_stats,
746            );
747
748            self.lost_count += lost_packets;
749
750            self.set_loss_detection_timer(handshake_status, now);
751
752            trace!("{} {:?}", trace_id, self);
753            return OnLossDetectionTimeoutOutcome {
754                lost_packets,
755                lost_bytes,
756            };
757        }
758
759        let epoch = if self.bytes_in_flight > 0 {
760            // Send new data if available, else retransmit old data. If neither
761            // is available, send a single PING frame.
762            let (_, e) = self.pto_time_and_space(handshake_status, now);
763
764            e
765        } else {
766            // Client sends an anti-deadlock packet: Initial is padded to earn
767            // more anti-amplification credit, a Handshake packet proves address
768            // ownership.
769            if handshake_status.has_handshake_keys {
770                packet::Epoch::Handshake
771            } else {
772                packet::Epoch::Initial
773            }
774        };
775
776        self.pto_count += 1;
777
778        let epoch = &mut self.epochs[epoch];
779
780        epoch.loss_probes = MAX_PTO_PROBES_COUNT.min(self.pto_count as usize);
781
782        // Skip packets that have already been acked or lost, and packets
783        // that don't contain either CRYPTO or STREAM frames and only return as
784        // many packets as the number of probe packets that will be sent.
785        let unacked_frames = epoch
786            .sent_packets
787            .iter_mut()
788            .filter_map(|p| {
789                if let SentStatus::Sent {
790                    has_data: true,
791                    frames,
792                    ..
793                } = &p.status
794                {
795                    Some(frames)
796                } else {
797                    None
798                }
799            })
800            .take(epoch.loss_probes)
801            .flatten()
802            .filter(|f| !matches!(f, frame::Frame::DatagramHeader { .. }));
803
804        // Retransmit the frames from the oldest sent packets on PTO. However
805        // the packets are not actually declared lost (so there is no effect to
806        // congestion control), we just reschedule the data they carried.
807        //
808        // This will also trigger sending an ACK and retransmitting frames like
809        // HANDSHAKE_DONE and MAX_DATA / MAX_STREAM_DATA as well, in addition
810        // to CRYPTO and STREAM, if the original packet carried them.
811        epoch.lost_frames.extend(unacked_frames.cloned());
812
813        self.pacer
814            .on_retransmission_timeout(!epoch.lost_frames.is_empty());
815
816        self.set_loss_detection_timer(handshake_status, now);
817
818        trace!("{} {:?}", trace_id, self);
819        OnLossDetectionTimeoutOutcome {
820            lost_packets: 0,
821            lost_bytes: 0,
822        }
823    }

826        &mut self, epoch: packet::Epoch, handshake_status: HandshakeStatus,
827        now: Instant,
828    ) {

830        self.bytes_in_flight = self

1002    fn send_quantum(&self) -> usize {
1003        let pacing_rate = self
1004            .pacer
1005            .pacing_rate(self.bytes_in_flight, &self.rtt_stats);
1006
1007        let floor = if pacing_rate < Bandwidth::from_kbits_per_second(1200) {
1008            self.max_datagram_size
1009        } else {
1010            2 * self.max_datagram_size
1011        };
1012
1013        pacing_rate
1014            .to_bytes_per_period(ReleaseDecision::EQUAL_THRESHOLD)
1015            .min(64 * 1024)
1016            .max(floor as u64) as usize
1017    }
