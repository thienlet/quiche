108impl RecoveryEpoch {
109    fn detect_and_remove_acked_packets(
110        &mut self, now: Instant, acked: &RangeSet, newly_acked: &mut Vec<Acked>,
111        rtt_stats: &RttStats, trace_id: &str,
112    ) -> AckedDetectionResult {
113        newly_acked.clear();
114
115        let mut acked_bytes = 0;
116        let mut spurious_losses = 0;
117        let mut spurious_pkt_thresh = None;
118        let mut has_ack_eliciting = false;
119        let mut has_in_flight_spurious_loss = false;
120
121        let largest_acked = self.largest_acked_packet.unwrap();
122
123        for ack in acked.iter() {
124            // Because packets always have incrementing numbers, they are always
125            // in sorted order.
126            let start = if self
127                .sent_packets
128                .front()
129                .filter(|e| e.pkt_num >= ack.start)
130                .is_some()
131            {
132                // Usually it will be the first packet.
133                0
134            } else {
135                self.sent_packets
136                    .binary_search_by_key(&ack.start, |p| p.pkt_num)
137                    .unwrap_or_else(|e| e)
138            };
139
140            for unacked in self.sent_packets.range_mut(start..) {
141                if unacked.pkt_num >= ack.end {
142                    break;
143                }
144
145                if unacked.time_acked.is_some() {
146                    // Already acked.
147                } else if unacked.time_lost.is_some() {
148                    // An acked packet was already declared lost.
149                    spurious_losses += 1;
150                    spurious_pkt_thresh
151                        .get_or_insert(largest_acked - unacked.pkt_num + 1);
152                    unacked.time_acked = Some(now);
153
154                    if unacked.in_flight {
155                        has_in_flight_spurious_loss = true;
156                    }
157                } else {
158                    if unacked.in_flight {
159                        self.in_flight_count -= 1;
160                        acked_bytes += unacked.size;
161                    }
162
163                    newly_acked.push(Acked {
164                        pkt_num: unacked.pkt_num,
165                        time_sent: unacked.time_sent,
166                        size: unacked.size,
167
168                        rtt: now.saturating_duration_since(unacked.time_sent),
169                        delivered: unacked.delivered,
170                        delivered_time: unacked.delivered_time,
171                        first_sent_time: unacked.first_sent_time,
172                        is_app_limited: unacked.is_app_limited,
173                    });
174
175                    trace!("{} packet newly acked {}", trace_id, unacked.pkt_num);
176
177                    self.acked_frames
178                        .extend(std::mem::take(&mut unacked.frames));
179
180                    has_ack_eliciting |= unacked.ack_eliciting;
181                    unacked.time_acked = Some(now);
182                }
183            }
184        }
185
186        self.drain_acked_and_lost_packets(now - rtt_stats.rtt());
187
188        AckedDetectionResult {
189            acked_bytes,
190            spurious_losses,
191            spurious_pkt_thresh,
192            has_ack_eliciting,
193            has_in_flight_spurious_loss,
194        }
195    }
196
197    fn detect_lost_packets(
198        &mut self, loss_delay: Duration, pkt_thresh: u64, now: Instant,
199        trace_id: &str, epoch: Epoch,
200    ) -> LossDetectionResult {
201        self.loss_time = None;
202
203        // Minimum time of kGranularity before packets are deemed lost.
204        let loss_delay = cmp::max(loss_delay, GRANULARITY);
205        let largest_acked = self.largest_acked_packet.unwrap_or(0);
206
207        // Packets sent before this time are deemed lost.
208        let lost_send_time = now.checked_sub(loss_delay).unwrap();
209
210        let mut lost_packets = 0;
211        let mut lost_bytes = 0;
212        let mut pmtud_lost_bytes = 0;
213
214        let mut largest_lost_pkt = None;
215
216        let unacked_iter = self.sent_packets
217        .iter_mut()
218        // Skip packets that follow the largest acked packet.
219        .take_while(|p| p.pkt_num <= largest_acked)
220        // Skip packets that have already been acked or lost.
221        .filter(|p| p.time_acked.is_none() && p.time_lost.is_none());
222
223        for unacked in unacked_iter {
224            // Mark packet as lost, or set time when it should be marked.
225            if unacked.time_sent <= lost_send_time ||
226                largest_acked >= unacked.pkt_num + pkt_thresh
227            {
228                self.lost_frames.extend(unacked.frames.drain(..));
229
230                unacked.time_lost = Some(now);
231
232                if unacked.pmtud {
233                    pmtud_lost_bytes += unacked.size;
234                    self.in_flight_count -= 1;
235
236                    // Do not track PMTUD probes losses.
237                    continue;
238                }
239
240                if unacked.in_flight {
241                    lost_bytes += unacked.size;
242
243                    // Frames have already been removed from the packet, so
244                    // cloning the whole packet should be relatively cheap.
245                    largest_lost_pkt = Some(unacked.clone());
246
247                    self.in_flight_count -= 1;
248
249                    trace!(
250                        "{} packet {} lost on epoch {}",
251                        trace_id,
252                        unacked.pkt_num,
253                        epoch
254                    );
255                }
256
257                lost_packets += 1;
258            } else {
259                let loss_time = match self.loss_time {
260                    None => unacked.time_sent + loss_delay,
261
262                    Some(loss_time) =>
263                        cmp::min(loss_time, unacked.time_sent + loss_delay),
264                };
265
266                self.loss_time = Some(loss_time);
267                break;
268            }
269        }
270
271        LossDetectionResult {
272            largest_lost_pkt,
273            lost_packets,
274            lost_bytes,
275            pmtud_lost_bytes,
276        }
277    }
278
279    fn drain_acked_and_lost_packets(&mut self, loss_thresh: Instant) {
280        // In order to avoid removing elements from the middle of the list
281        // (which would require copying other elements to compact the list),
282        // we only remove a contiguous range of elements from the start of the
283        // list.
284        //
285        // This means that acked or lost elements coming after this will not
286        // be removed at this point, but their removal is delayed for a later
287        // time, once the gaps have been filled.
288        while let Some(pkt) = self.sent_packets.front() {
289            if let Some(time_lost) = pkt.time_lost {
290                if time_lost > loss_thresh {
291                    break;
292                }
293            }
294
295            if pkt.time_acked.is_none() && pkt.time_lost.is_none() {
296                break;
297            }
298
299            self.sent_packets.pop_front();
300        }
301    }
302}

380    fn loss_time_and_space(&self) -> (Option<Instant>, packet::Epoch) {
381        let mut epoch = packet::Epoch::Initial;
382        let mut time = self.epochs[epoch].loss_time;
383
384        // Iterate over all packet number spaces starting from Handshake.
385        for e in [packet::Epoch::Handshake, packet::Epoch::Application] {
386            let new_time = self.epochs[e].loss_time;
387            if time.is_none() || new_time < time {
388                time = new_time;
389                epoch = e;
390            }
391        }
392
393        (time, epoch)
394    }
395
396    fn pto_time_and_space(
397        &self, handshake_status: HandshakeStatus, now: Instant,
398    ) -> (Option<Instant>, packet::Epoch) {
399        let mut duration = self.pto() * 2_u32.pow(self.pto_count);
400
401        // Arm PTO from now when there are no inflight packets.
402        if self.bytes_in_flight == 0 {
403            if handshake_status.has_handshake_keys {
404                return (Some(now + duration), packet::Epoch::Handshake);
405            } else {
406                return (Some(now + duration), packet::Epoch::Initial);
407            }
408        }
409
410        let mut pto_timeout = None;
411        let mut pto_space = packet::Epoch::Initial;
412
413        // Iterate over all packet number spaces.
414        for e in [
415            packet::Epoch::Initial,
416            packet::Epoch::Handshake,
417            packet::Epoch::Application,
418        ] {
419            let epoch = &self.epochs[e];
420            if epoch.in_flight_count == 0 {
421                continue;
422            }
423
424            if e == packet::Epoch::Application {
425                // Skip Application Data until handshake completes.
426                if !handshake_status.completed {
427                    return (pto_timeout, pto_space);
428                }
429
430                // Include max_ack_delay and backoff for Application Data.
431                duration +=
432                    self.rtt_stats.max_ack_delay * 2_u32.pow(self.pto_count);
433            }
434
435            let new_time = epoch
436                .time_of_last_ack_eliciting_packet
437                .map(|t| t + duration);
438
439            if pto_timeout.is_none() || new_time < pto_timeout {
440                pto_timeout = new_time;
441                pto_space = e;
442            }
443        }
444
445        (pto_timeout, pto_space)
446    }
447
448    fn set_loss_detection_timer(
449        &mut self, handshake_status: HandshakeStatus, now: Instant,
450    ) {
451        let (earliest_loss_time, _) = self.loss_time_and_space();
452
453        if let Some(to) = earliest_loss_time {
454            // Time threshold loss detection.
455            self.loss_timer.update(to);
456            return;
457        }
458
459        if self.bytes_in_flight == 0 && handshake_status.peer_verified_address {
460            self.loss_timer.clear();
461            return;
462        }
463
464        // PTO timer.
465        if let (Some(timeout), _) = self.pto_time_and_space(handshake_status, now)
466        {
467            self.loss_timer.update(timeout);
468        }
469    }
470
471    fn detect_lost_packets(
472        &mut self, epoch: packet::Epoch, now: Instant, trace_id: &str,
473    ) -> (usize, usize) {
474        let loss_delay = cmp::max(self.rtt_stats.latest_rtt, self.rtt())
475            .mul_f64(self.time_thresh);
476
477        let loss = self.epochs[epoch].detect_lost_packets(
478            loss_delay,
479            self.pkt_thresh,
480            now,
481            trace_id,
482            epoch,
483        );
484
485        if let Some(pkt) = loss.largest_lost_pkt {
486            if !self.congestion.in_congestion_recovery(pkt.time_sent) {
487                (self.congestion.cc_ops.checkpoint)(&mut self.congestion);
488            }
489
490            (self.congestion.cc_ops.congestion_event)(
491                &mut self.congestion,
492                self.bytes_in_flight,
493                loss.lost_bytes,
494                &pkt,
495                now,
496            );
497
498            self.bytes_in_flight -= loss.lost_bytes;
499        };
500
501        self.bytes_in_flight -= loss.pmtud_lost_bytes;
502
503        self.epochs[epoch]
504            .drain_acked_and_lost_packets(now - self.rtt_stats.rtt());
505
506        self.congestion.lost_count += loss.lost_packets;
507
508        (loss.lost_packets, loss.lost_bytes)
509    }
510}

551    fn on_packet_sent(
552        &mut self, mut pkt: Sent, epoch: packet::Epoch,
553        handshake_status: HandshakeStatus, now: Instant, trace_id: &str,
554    ) {
555        let ack_eliciting = pkt.ack_eliciting;
556        let in_flight = pkt.in_flight;
557        let sent_bytes = pkt.size;
558
559        if ack_eliciting {
560            self.outstanding_non_ack_eliciting = 0;
561        } else {
562            self.outstanding_non_ack_eliciting += 1;
563        }
564
565        if in_flight && ack_eliciting {
566            self.epochs[epoch].time_of_last_ack_eliciting_packet = Some(now);
567        }
568
569        self.congestion.on_packet_sent(
570            self.bytes_in_flight,
571            sent_bytes,
572            now,
573            &mut pkt,
574            &self.rtt_stats,
575            self.bytes_lost,
576            in_flight,
577        );
578
579        if in_flight {
580            self.epochs[epoch].in_flight_count += 1;
581            self.bytes_in_flight += sent_bytes;
582
583            self.set_loss_detection_timer(handshake_status, now);
584        }
585
586        self.bytes_sent += sent_bytes;
587
588        self.epochs[epoch].sent_packets.push_back(pkt);
589
590        trace!("{} {:?}", trace_id, self);
591    }

598    #[allow(clippy::too_many_arguments)]
599    fn on_ack_received(
600        &mut self, ranges: &ranges::RangeSet, ack_delay: u64,
601        epoch: packet::Epoch, handshake_status: HandshakeStatus, now: Instant,
602        trace_id: &str,
603    ) -> OnAckReceivedOutcome {
604        let largest_acked = ranges.last().unwrap();
605
606        // Update the largest acked packet.
607        let largest_acked = self.epochs[epoch]
608            .largest_acked_packet
609            .unwrap_or(0)
610            .max(largest_acked);
611
612        self.epochs[epoch].largest_acked_packet = Some(largest_acked);
613
614        let AckedDetectionResult {
615            acked_bytes,
616            spurious_losses,
617            spurious_pkt_thresh,
618            has_ack_eliciting,
619            has_in_flight_spurious_loss,
620        } = self.epochs[epoch].detect_and_remove_acked_packets(
621            now,
622            ranges,
623            &mut self.newly_acked,
624            &self.rtt_stats,
625            trace_id,
626        );
627
628        self.lost_spurious_count += spurious_losses;
629        if let Some(thresh) = spurious_pkt_thresh {
630            self.pkt_thresh =
631                self.pkt_thresh.max(thresh.min(MAX_PACKET_THRESHOLD));
632        }
633
634        // Undo congestion window update.
635        if has_in_flight_spurious_loss {
636            (self.congestion.cc_ops.rollback)(&mut self.congestion);
637        }
638
639        if self.newly_acked.is_empty() {
640            return OnAckReceivedOutcome::default();
641        }
642
643        // Check if largest packet is newly acked.
644        let largest_newly_acked = self.newly_acked.last().unwrap();
645
646        if largest_newly_acked.pkt_num == largest_acked && has_ack_eliciting {
647            let latest_rtt = now - largest_newly_acked.time_sent;
648            self.rtt_stats.update_rtt(
649                latest_rtt,
650                Duration::from_micros(ack_delay),
651                now,
652                handshake_status.completed,
653            );
654        }
655
656        // Detect and mark lost packets without removing them from the sent
657        // packets list.
658        let (lost_packets, lost_bytes) =
659            self.detect_lost_packets(epoch, now, trace_id);
660
661        self.congestion.on_packets_acked(
662            self.bytes_in_flight,
663            &mut self.newly_acked,
664            &self.rtt_stats,
665            now,
666        );
667
668        self.bytes_in_flight -= acked_bytes;
669
670        self.pto_count = 0;
671
672        self.set_loss_detection_timer(handshake_status, now);
673
674        self.epochs[epoch]
675            .drain_acked_and_lost_packets(now - self.rtt_stats.rtt());
676
677        OnAckReceivedOutcome {
678            lost_packets,
679            lost_bytes,
680            acked_bytes,
681            spurious_losses,
682        }
683    }

685    fn on_loss_detection_timeout(
686        &mut self, handshake_status: HandshakeStatus, now: Instant,
687        trace_id: &str,
688    ) -> OnLossDetectionTimeoutOutcome {
689        let (earliest_loss_time, epoch) = self.loss_time_and_space();
690
691        if earliest_loss_time.is_some() {
692            // Time threshold loss detection.
693            let (lost_packets, lost_bytes) =
694                self.detect_lost_packets(epoch, now, trace_id);
695
696            self.set_loss_detection_timer(handshake_status, now);
697
698            trace!("{} {:?}", trace_id, self);
699            return OnLossDetectionTimeoutOutcome {
700                lost_packets,
701                lost_bytes,
702            };
703        }
704
705        let epoch = if self.bytes_in_flight > 0 {
706            // Send new data if available, else retransmit old data. If neither
707            // is available, send a single PING frame.
708            let (_, e) = self.pto_time_and_space(handshake_status, now);
709
710            e
711        } else {
712            // Client sends an anti-deadlock packet: Initial is padded to earn
713            // more anti-amplification credit, a Handshake packet proves address
714            // ownership.
715            if handshake_status.has_handshake_keys {
716                packet::Epoch::Handshake
717            } else {
718                packet::Epoch::Initial
719            }
720        };
721
722        self.pto_count += 1;
723
724        let epoch = &mut self.epochs[epoch];
725
726        epoch.loss_probes =
727            cmp::min(self.pto_count as usize, MAX_PTO_PROBES_COUNT);
728
729        let unacked_iter = epoch.sent_packets
730            .iter_mut()
731            // Skip packets that have already been acked or lost, and packets
732            // that don't contain either CRYPTO or STREAM frames.
733            .filter(|p| p.has_data && p.time_acked.is_none() && p.time_lost.is_none())
734            // Only return as many packets as the number of probe packets that
735            // will be sent.
736            .take(epoch.loss_probes);
737
738        // Retransmit the frames from the oldest sent packets on PTO. However
739        // the packets are not actually declared lost (so there is no effect to
740        // congestion control), we just reschedule the data they carried.
741        //
742        // This will also trigger sending an ACK and retransmitting frames like
743        // HANDSHAKE_DONE and MAX_DATA / MAX_STREAM_DATA as well, in addition
744        // to CRYPTO and STREAM, if the original packet carried them.
745        for unacked in unacked_iter {
746            epoch.lost_frames.extend_from_slice(&unacked.frames);
747        }
748
749        self.set_loss_detection_timer(handshake_status, now);
750
751        trace!("{} {:?}", trace_id, self);
752
753        OnLossDetectionTimeoutOutcome {
754            lost_packets: 0,
755            lost_bytes: 0,
756        }
757    }
758
759    fn on_pkt_num_space_discarded(
760        &mut self, epoch: packet::Epoch, handshake_status: HandshakeStatus,
761        now: Instant,
762    ) {
763        let epoch = &mut self.epochs[epoch];
764
765        let unacked_bytes = epoch
766            .sent_packets
767            .iter()
768            .filter(|p| {
769                p.in_flight && p.time_acked.is_none() && p.time_lost.is_none()
770            })
771            .fold(0, |acc, p| acc + p.size);
772
773        self.bytes_in_flight -= unacked_bytes;
774
775        epoch.sent_packets.clear();
776        epoch.lost_frames.clear();
777        epoch.acked_frames.clear();
778
779        epoch.time_of_last_ack_eliciting_packet = None;
780        epoch.loss_time = None;
781        epoch.loss_probes = 0;
782        epoch.in_flight_count = 0;
783
784        self.set_loss_detection_timer(handshake_status, now);
785    }

849        // Congestion Window is updated only when it's not updated already.
850        // Update cwnd if it hasn't been updated yet.
851        if self.cwnd() ==
852            self.max_datagram_size *
853                self.congestion.initial_congestion_window_packets
854        {
855            self.congestion.congestion_window = new_max_datagram_size *
856                self.congestion.initial_congestion_window_packets;
857        }
858
859        self.congestion.pacer = pacer::Pacer::new(
860            self.congestion.pacer.enabled(),
861            self.cwnd(),
862            0,
863            new_max_datagram_size,
864            self.congestion.pacer.max_pacing_rate(),
865        );
866
867        self.max_datagram_size = new_max_datagram_size;
868    }

995        write!(f, "timer={:?} ", self.loss_timer)?;
996        write!(f, "latest_rtt={:?} ", self.rtt_stats.latest_rtt)?;
997        write!(f, "srtt={:?} ", self.rtt_stats.smoothed_rtt)?;
998        write!(f, "min_rtt={:?} ", *self.rtt_stats.min_rtt)?;
999        write!(f, "rttvar={:?} ", self.rtt_stats.rttvar)?;
1000        write!(f, "cwnd={} ", self.cwnd())?;
1001        write!(f, "ssthresh={} ", self.congestion.ssthresh)?;
1002        write!(f, "bytes_in_flight={} ", self.bytes_in_flight)?;
1003        write!(f, "app_limited={} ", self.congestion.app_limited)?;
1004        write!(
1005            f,
1006            "congestion_recovery_start_time={:?} ",
1007            self.congestion.congestion_recovery_start_time
1008        )?;
1009        write!(f, "{:?} ", self.congestion.delivery_rate)?;
1010        write!(f, "pacer={:?} ", self.congestion.pacer)?;
1011
1012        if self.congestion.hystart.enabled() {
1013            write!(f, "hystart={:?} ", self.congestion.hystart)?;
1014        }
1015
1016        // CC-specific debug info
1017        (self.congestion.cc_ops.debug_fmt)(&self.congestion, f)?;
1018
1019        Ok(())
1020    }
